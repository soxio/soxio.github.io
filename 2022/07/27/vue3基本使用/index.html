数据

reactive(复杂数据类型)

<template>
  {{ obj }}
  <button @click="changeMsg">改变</button>
</template>

<script lang="ts">
import { defineComponent, reactive } from 'vue'

export default defineComponent({
  setup() {
    let obj = reactive({
      name: '张三',
      age: 26
    })
    const changeMsg = () => {
      obj.name = '李四'
    }
    return {
      obj,
      changeMsg
    }
  }
})
</script>

ref(基本和复杂数据类型)

在script标签中需要用变量.value来访问变量值

<template>
  <h1>{{ msg }}</h1>
  <button @click="changeMsg">改变</button>
</template>

<script lang="ts">
import { defineComponent, ref } from 'vue'

export default defineComponent({
  setup() {
    let msg = ref('张三')
    const changeMsg = () => {
      msg.value = '李四'
    }
    return {
      msg,
      changeMsg
    }
  }
})
</script>

生命周期

生命周期 vue3 vue2
创建实例前 setup beforeCreate
创建实例后 steup created
挂载DOM前 onBeforeMount beforeMount
挂载DOM后 onMounted mounted
更新组件前 onBeforeUpdate beforeUpdate
更新组件后 onUpdated updated
销毁组件前 onBeforeUnmount beforeDestroy
销毁组件后 onUnmounted destroyed

方法

setup★

使用setup语法糖无需将定义的变量和函数return出去，只需将变量和函数定在setup函数的顶层

<template>
  <h2>{{ count }}</h2>
  <button @click="change">增加</button>
</template>

<script setup lang="ts">
import { ref } from 'vue';

  const count = ref(0)
  const change = () => {
    count.value++
  }
</script>

computed 计算属性

<template>
  <div>现在年龄 <input type="text" v-model.number="age"/></div>
  <div>明年年龄 {{ nextAge }}</div>
  <div>后年年龄 <input type="text" v-model.number="nextAge2"></div>
</template>

<script setup lang="ts">
import { computed } from '@vue/reactivity';
import { ref } from 'vue';

const age = ref(25)

const nextAge = computed(() => {
  return age.value + 1
})

const nextAge2 = computed({
  get() {
    return age.value + 2
  },
  // 计算属性影响原值
  set(val: number) {
    age.value = val - 2
  }
})
</script>

watch 侦听器

watch 侦听器, 接收三个参数

  1. 参数1: 监视的数据源
  2. 参数2: 回调函数
  3. 参数3: 额外的配置

监听单个值

<template>
  <h2>{{ money }}</h2>
  <button @click="add">+</button>
</template>

<script setup lang="ts">
import { ref, watch } from 'vue';

const money = ref(0)
const add = () => {
  money.value++
}

watch(money, (newVal, oldVal) => {
  console.log('新值', newVal, '旧值', oldVal);

})
</script>

监听多个值

<template>
  <h2>{{ money }}</h2>
  <h2>{{ count }}</h2>
  <button @click="add">+</button>
</template>

<script setup lang="ts">
import { ref, watch } from 'vue';

const money = ref(0)
const count = ref(0)
const add = () => {
  money.value++
  count.value++
}

watch([money, count], (newVal, oldVal) => {
  console.log('新值', newVal, '旧值', oldVal);

})
</script>

监听配置项

<template>
  <h2>{{ people.name }}</h2>
  <button @click="change">改变</button>
</template>

<script setup lang="ts">
import { ref, watch } from 'vue';

  const people = ref({
    name: '张三',
    age: 26
  })
  const change = () => {
    people.value.name = '李四'
  }
  watch(people, (newVal, oldVal) => {
    console.log('新值', newVal, '旧值', oldVal);
    
  }, {
    immediate: true, // 组件首次加载触发
    deep: true // 对象深度监听
  })
</script>

监听对象属性

<template>
  <h2>{{ people.name }}</h2>
  <h2>{{ people.age }}</h2>
  <button @click="change">改变名字</button>
  <button @click="people.age += 1">改变年龄</button>
</template>

<script lang="ts" setup>
import { ref, watch } from 'vue';

const people = ref({
  name: '张三',
  age: 26
})
const change = () => {
  people.value.name = '李四'
}
watch(() => {
  return people.value.name
}, () => {
  console.log('对象的名字发生了变化');

})
</script>

组件通讯

defineProps 父传子

  • 父组件
<template>
  <h1>父组件</h1>
  <Son :name="'兰博基尼'" :money="50000"/>
</template>

<script setup lang="ts">
import Son from './components/Son.vue';
</script>
  • 子组件
<template>
  <h2>子组件</h2>
  <div>汽车：{{ name }}</div>
  <div>价格：{{ money }}$</div>
</template>

<script setup lang="ts">
  interface CarType {
    name: string,
    money: number
  }
  const props = defineProps<CarType>()
</script>

defineEmit 子传父

  • 父组件
<template>
  <h1>父组件：{{ msg }}</h1>
  <Son @send="receive"/>
</template>

<script setup lang="ts">
import { ref } from 'vue';
import Son from './components/Son.vue';
const msg = ref('')
const receive = (val: string) => {
  msg.value = val
}
</script>
  • 子组件
<template>
  <button @click="sendData">发送给父组件数据</button>
</template>

<script setup lang="ts">
// 定义emit
const emit = defineEmits(['send'])
const sendData = () => {
  // 使用emit
  emit('send', '子组件的数据')
}
</script>

provide / inject 依赖注入

  • 父组件
provide('money', ref(10000))
  • 后代组件
const money = inject<Ref<number>>('str')
console.log(money);

获取实例对象（配合defineExpose）

  • 父组件
<template>
  <Son ref="son"/>
</template>

<script setup lang="ts">
import Son from './components/Son.vue';
import { onMounted, ref } from 'vue';

const son = ref<InstanceType<typeof Son>>()
onMounted(() => {
  console.log(son.value);
  
})

</script>
  • 子组件
<template>
  <h1>{{ msg }}</h1>
</template>

<script setup lang="ts">
const msg = '我是子组件'
// 将需要的数据暴露出去
defineExpose({
  msg
})
</script>

vue3中使用函数替代

toRefs

<template>
  <h2>{{ user.name }}</h2>
  <h2>{{ name }}</h2>
  <button @click="user.name += '~'">改变名字</button>
</template>

<script setup lang="ts">
import { ref, toRefs } from 'vue';

const user = ref({
  name: '张三',
  age: 26
})
// 解构出来的name 失去响应式
// const { name } = user.value
// 使用refs 保持响应式
const { name } = toRefs(user.value)
</script>

安装

npm install vue-router

使用导航式路由

  • 新建router/index.ts
import { createRouter, createWebHashHistory, createWebHistory } from 'vue-router'


const router = createRouter({
  // hash 模式
  // history: createWebHashHistory(),
  // history 模式
  history: createWebHistory(),
  routes: [
    {
      path: '/home',
      component: () => import('@/components/Home.vue')
    },
    {
      path: '/login',
      component: () => import('@/components/Login.vue')
    },
  ]
})

export default router
  • main.ts引入
import { createApp } from 'vue'
import App from './App.vue'
import router from '@/router'

createApp(App).use(router).mount('#app')

pinia 状态管理

安装

npm install pinia

使用

  • main.ts
import { createApp } from 'vue'
import App from './App.vue'
import { createPinia } from 'pinia'

const pinia = createPinia()
createApp(App).use(pinia).mount('#app')
  • store/counter.ts
import { defineStore } from 'pinia'

const useCounterStore = defineStore('count', {
  state() {
    return {
      count: 0
    }
  },
  // 支持同步和异步操作
  actions: {
    add() {
      this.count++
    }
  },
  getters: {
    double(): number {
      return this.count * 2
    }
  }
})

export default useCounterStore
  • App.vue
<template>
  <h2>{{ counter.count }}</h2>
  <h2>{{ counter.double }}</h2>
  <button @click="counter.add">+</button>
</template>

<script setup lang="ts">
import useCounterStore from './store/counter';

const counter = useCounterStore()
console.log(counter.count);

</script>

storeToRefs 保持响应式

<template>
  <h2>{{ counter.count }}</h2>
  <h2>{{ count }}</h2>
  <button @click="counter.add">+</button>
</template>

<script setup lang="ts">
import useCounterStore from './store/counter';
import { storeToRefs } from 'pinia'
    
const counter = useCounterStore()
const { count } = storeToRefs(counter)
console.log(counter.count);

</script>
