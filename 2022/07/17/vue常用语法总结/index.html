watch

export default {
  data() {
    return {
      screenHeight: null
    }
  },
  watch: {
    // 方式一 适用与简单数据类型
    // screenHeight(newVal) {
    //   console.log(newVal);
    // },
    // 方式二 适用与简单和复杂数据类型
    screenHeight: {
      handler(val) {
        console.log(val);
      },
      deep: true, // 深度监听，复杂数据类型需要开启
      immediate: true // 初始化监听
    }
  },
  mounted() {
    // 监听屏幕分辨率事件
    window.addEventListener('resize', () => {
      this.screenHeight = screen.height
    })
  },
}

动态绑定class

方式1 对象形式

:class="{'red': current === index}"

方式2 三元表达式

:class="current === index ? 'red':''"

动态绑定style

方式1 对象形式

<div :style="{ color: 'red' }">哈哈</div>

方式2 三元表达式

<div :style="{ color: (current > 0 ? 'red' : 'green') }">哈哈</div>

方式3 绑定data数据

:style="styleObject"
data() {
    return{
      styleObject: {
        color: 'red',
        fontSize: '13px'
      }  
    }
}
