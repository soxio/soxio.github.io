基本数据类型

// 数字类型
let num: number = 123

// 字符串类型
let str: string = '123'

// 布尔类型
let flag: boolean = true

// null类型
let n: null = null

// undefined类型
let u: undefined = undefined

复杂数据类型

数组

// 写法一（推荐）
let arr: number[] = [1, 2, 3]

// 写法二
let arr2: Array<number> = [1, 2, 3]

对象

// 对象类型 写法一
let person: { name: string, age: number, sayHi(str: string): void } = {
  name: '张三',
  age: 28,
  sayHi: function(str) {
    console.log(str);
    
  }
}

// 写法二推荐
type Person = { name: string, age: number, sayHi: (str: string) => void }

let person1: Person = {
  name: '李四',
  age: 27,
  sayHi: function(str) {
    console.log(str); 
  }
}

函数

// 写法一
function fn(name: string, age: number): void {
  console.log(name, age);

}

// 写法二
const fn2 = (name: string, age: number): void => {
  console.log('执行某个业务');

}

联合类型

let bg: number | string = '2'

let arr: (number | string)[] = [1, 2, '3']

type 自定义类型

  • 使用
// 数据自定义类型
type MyType = (number | string)[]
let arr: MyType = [1, 2, '3']

// 函数自定义类型
type AddType = (a: number, b: number) => number
const add: AddType = (a, b) => a + b
  • 类似继承
type Point2D  = {
  x: number,
  y: number
}

type Point3D = Point2D & {
  z: number
}

let t1: Point3D = {
  x: 1,
  y: 1,
  z: 1
}

void

// 函数无需返回值
const fn = (): void => {
  console.log('执行某个业务');
}

const a = fn()
console.log('a', a); // undefined

可选参数

// age为可选参数 必传参数必须在前面
const fn = (name: string, age?: number): void => {
  console.log('执行某个业务');
}

类型推论

// 变量age的类型被自动推断为：number
let age = 18

// 函数形参设置了默认值和返回值也能自动推论
function add(num1 = 0, num2 = 0) {
  return num1 + num2
}

interface 接口类型（只能用于对象）

  • 使用
interface IPerson {
  name: string,
  age: number,
  sayHi(): void
}

let obj: IPerson = {
  name: '小明',
  age: 25,
  sayHi() {
    console.log('业务代码');
  }
}
  • 继承（减少重复项）
interface Point2D {
  x: number,
  y: number
}

interface Point3D extends Point2D {
  z: number
}

let point: Point3D = {
  x: 1,
  y: 2,
  z: 3
}

特殊的数组，限制类型和长度

let position: [number, number] = [1, 2]

字面量类型（单选）

type Method = 'GET' | 'POST' | 'PUT' | 'DELETE'
const m: Method = 'GET'

枚举类型

注意：枚举成员是有值的，默认从0开始自增的数值

// 定义
enum Gender {
  Boy = 0,
  Girl = 1,
  Unknown = 2
}
// 使用
const query = {
  name: '张三',
  gender: Gender.Boy
}

类型断言

// 主动确定类型
const alink = document.querySelector('#a') as HTMLAnchorElement
alink.href = 'www.baidu.com'

泛型

基本使用

// 传入什么类型就返回什么类型
const fn = <T>(value: T): T => {
  return value
}

fn(1).toFixed()
fn('1').trim()

泛型接口

interface Res<T> {
  code: number,
  msg: string,
  result: T
}

let res: Res<number[]> = {
  code: 200,
  msg: '成功',
  result: [1, 2, 3]
}

interface Todo {
  id: number,
  content: string,
  done: boolean
}

let res2: Res<Todo[]> = {
  code: 200,
  msg: '成功',
  result: [
    { id: 1, content: '吃饭', done: false },
    { id: 2, content: '玩游戏', done: false },
  ]
}

泛型约束

// 约束函数的参数必须要有 length 属性
const getLength = <T extends { length: number }>(value: T): number => {
  return value.length
}

getLength({ length: 0 })

// 可以根据对象的key只能匹配
const getProp = <T extends object, K extends keyof T>(obj: T, key: K) => {
    return obj[key]
}

getProp({name: 'jack', age: 18}, 'age')

多个泛型

泛型可以设置默认值，默认值可以被覆盖

const fn2 = <T, K = string>(a: T, b: K) => {
  console.log(a, b);
}

fn2('哈哈', 11)
fn2(11, '哈哈')

获取对象key名称

type Hero = {
  name: string,
  age: number
}

type Keys = keyof Hero

let a:Keys = 'name'
