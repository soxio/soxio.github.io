安装

npm install vuex

配置

index.js

  • 在src目录下新建store文件夹，store文件夹新建index.js

    import Vue from "vue"
import Vuex from "vuex"

// 1. 安装插件
Vue.use(Vuex)


// 2. 创建对象
const store = new Vuex.Store({
  state: {
    counter: 1000
  }
})



// 3. 导出store对象
export default store

main.js

  • 导入store对象并挂载
// 1. 导入store对象
import store from './store/index'
new Vue({
  // 2. 挂载store对象
  store,
  router,
  render: h => h(App),
}).$mount('#app')

状态管理

state（数据）

直接使用state数据

推荐使用Vue.set和Vue.delete对数组和对象进行操作，这样操作得到的数据是响应式的

<h1>{{$store.state.counter}}</h1>

mutation（方法）

修改state数据

推荐只进行同步操作，进行异步操作的数据无法被devtools跟踪到

  • 在 index.js 里通过mutations增加操作state的方法
const store = new Vuex.Store({
  state: {
    counter: 1000
  },
  // 通过mutations属性 定义方法来操控state的数据，这样做的好处是可以记录每一次的操作
  mutations: {
    add(state) {
      state.counter++
    },
    sub(state) {
      state.counter--
    }
  },
  actions: {

  },
  getters: {

  },
  modules: {

  }
})
  • 在组件里增加事件方法，通过commit 来调用 mutations 里的方法
<template>
  <div>
    <h1>{{$store.state.counter}}</h1>
    <button @click="addition">+</button>
    <button @click="subtraction">-</button>
  </div>
</template>

<script>
export default {
  methods: {
    addition() {
      this.$store.commit('add')
    },
    subtraction() {
      this.$store.commit('sub')
    }
  },
}
</script>

getters（计算属性）

基本使用

一般 getters 都是放在根模块最外层, 方便获取所有子模块的state, 从而统一简化数据的渲染

  • 定义
const store = new Vuex.Store({
  ...
  getters: {
	// 返回counter的平方
	powerCounter(state) {
	  return state.counter * state.counter
	}
  ...
  }
})
  • 使用
<h1>{{$store.getters.powerCounter}}</h1>

带参使用

  • 定义 返回自定义年龄段的学生数组
const store = new Vuex.Store({
  ...
  getters: {
  	// 接受用户自定义 返回的是一个函数
    anyAge(state) {
      return (age) => {
        return state.list.filter(item => item.age > age)
      }
    }
  ...
  }
})
  • 使用
<h1>{{$store.getters.anyAge(20)}}</h1>

actions（异步操作）

建议所有的异步操作都放在actions里进行

  • index.js
...
state: {
    info: {
      name: '张三',
      age: 18,
      gender: '男',
      hobby: '干饭'
    }
  },
...
...
mutations: {
 updateInfo(state) {
      Vue.set(state.info, 'adress', '广东')
    }, 
}
...
actions: {
   // 1. 解决异步操作无法被 vue-devtools追踪的问题
   asyncUpdate(context, payload) {
     // 无参
     // setTimeout(() => {
     //   context.commit('updateInfo')
     // }, 1000)

     // 有参
     return new Promise((resolve, reject) => {
       setTimeout(() => {
         context.commit('updateInfo')
       }, 1000)
       console.log(payload);
       resolve('我是结果')
     })
   }
 },
  • app.vue
// 异步修改
   asyncUpdate() {
     // 1. 无参
     // this.$store.dispatch('asyncUpdate')

     // 2. 有参
     this.$store.dispatch('asyncUpdate', '我是参数').then(res => {
       console.log(res);
     })

   }

modules（模块）

  • 由于使用单一状态树，应用的所有状态会集中到一个比较大的对象。当应用变得非常复杂时，store 对象就有可能变得相当臃肿。

    为了解决以上问题，Vuex 允许我们将 store 分割成模块（module）。每个模块拥有自己的 state、mutation、action、getter、甚至是嵌套子模块——从上至下进行同样方式的分割：

const moduleA = {
  state: () => ({ ... }),
  mutations: { ... },
  actions: { ... },
  getters: { ... }
}

const moduleB = {
  state: () => ({ ... }),
  mutations: { ... },
  actions: { ... }
}

const store = new Vuex.Store({
  modules: {
    a: moduleA,
    b: moduleB
  }
})

store.state.a // -> moduleA 的状态
store.state.b // -> moduleB 的状态

锁定命名空间（默认状态下, mutations / actions 是全局通用）

modules: {
   user: {
     namespaced: true,
     state: {
       name: '张三'
     },
     mutations: {
       updateModules(state) {
         state.name = '李四'
       }
     }
   }
 }

提交命名空间的三种方式

  • 方式一
<h1>-----修改模块中被锁定的数据-------</h1>
  {{$store.state.user.name}}
<button @click="updateModules">修改</button>
methods: {
  updateModules() {
     // 加上 user/ 限定名
     this.$store.commit('user/updateModules')
   }
}
  • 方式二
<h1>-----修改模块中被锁定的数据-------</h1>
  {{$store.state.user.name}}
<button @click="updateModules">修改</button>
import { mapMutations } from "vuex";
...
methods: {
  // 在辅助函数中通过第一个参数限定命名空间
  ...mapMutations('user', ['updateModules'])
}
  • 方式三
<h1>-----修改模块中被锁定的数据-------</h1>
  {{$store.state.user.name}}
<button @click="updateModules">修改</button>
import { createNamespacedHelpers } from "vuex";
const { mapMutations } = createNamespacedHelpers("user");
...
methods: {
   ...mapMutations(['updateModules'])
 }
