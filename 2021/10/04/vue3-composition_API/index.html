vue3-composition_API 基本使用

vue3-vue3compositionAPI

组合式API

龙应该藏在云里

setup

  • setup 在 beforeCreate钩子之前执行
  • setup 中无法使用this
  • 变量和函数需要在setup中返回

参数

  • props 用于使用来自父组件传递的参数
  • context 包含三个属性
    • attrs : 所有非prop的attribute
    • slots : 父组件传递过来的插槽
    • emit : 组件需要发出事件时使用
setup(props, context) {}

返回值

  • 可存放data和method

生命周期

  • vue2
  • beforeCreate
    • created
    • beforeMount
    • mounted
    • beforeUpdate
    • updated
    • beforeDestroy
    • destroyed
  • vue3
    • setup 创建实例前
    • onBeforeMount 挂载DOM前
    • onMounted 挂载DOM后
    • onBeforeUpdate 更新组件前
    • onUpdated 更新组件后
    • onBeforeUnmount 卸载销毁前
    • onUnmounted 卸载销毁后

state

reactive（复杂数据类型）

<template>
  <h3>{{state.counter}}</h3>
  <div><button @click="add">+1</button></div>
</template>

<script>
import { reactive } from 'vue'
export default {
  setup() {
    // reactive 定义数据 必须传入对象或数组
    let state = reactive({
      counter: 0
    })
    // 定义事件函数，需要在setup中return
    const add = () => {
      state.counter++
    }

    return {
      state,
      add
    }
  }

}
</script>

ref（简单/复杂数据类型）

定义数据修改需要.value，模板则不需要

简单数据类型

<template>
  <!-- 自动解构 无需写 counter.value -->
  <h3>{{counter}}</h3>
  <div><button @click="add">+1</button></div>
</template>

<script>
import { ref } from 'vue'
export default {
  setup() {
    // ref 定义数据 推荐使用
    let counter = ref(0)
    // 定义事件函数，需要在setup中return
    const add = () => {
      // setup中无法自动解构
      counter.value++
    }

    return {
      counter,
      add
    }
  }

}
</script>

复杂数据类型

<template>
  <h3>{{obj.name}}</h3>
  <h3>{{obj.age}}</h3>
  <div><button @click="change">修改名称</button></div>
</template>

<script>
import { reactive, ref, toRefs } from 'vue'
export default {
  setup() {
    const obj = ref({
      name: '张三',
      age: 18
    })


    const change = () => {
      obj.value.name = '李四'
    }

    return {
      obj,
      change,
      
    }
  }

}
</script>

toRef（解构单个对象属性）

用于解构出对象的属性，属性仍然和对象有关联

<template>
  <h3>{{obj.name}}</h3>
  <h3>{{name}}</h3>
  <div><button @click="change">修改名称</button></div>
</template>

<script>
import { reactive, toRef } from 'vue'
export default {
  setup() {
    const obj = reactive({
      name: '张三',
      age: 18
    })

    // 解构出name属性 这时的name仍然是关联obj的
    let name = toRef(obj, 'name')

    const change = () => {
      // 解构出的name需要用.value修改
      name.value = '李四'
    }

    return {
      obj,
      change,
      name
    }
  }

}
</script>

toRefs（解构全部对象属性）

将响应式对象转换为普通对象，其中结果对象的每个 property 都是指向原始对象相应 property 的 ref

<template>
  <h3>{{name}}</h3>
  <h3>{{age}}</h3>
  <div><button @click="change">修改名称</button></div>
</template>

<script>
import { reactive, toRefs } from 'vue'
export default {
  setup() {
    const obj = reactive({
      name: '张三',
      age: 18
    })

    let { name, age } = toRefs(obj)

    const change = () => {
      // 解构出的name需要用.value修改
      name.value = '李四'
      age.value = 20
    }

    return {
      name, 
      age,
      change,
      
    }
  }

}
</script>

computed

传入函数

<template>
  <h3>今年{{age}}岁</h3>
  <h3>明年{{newAge}}岁</h3>
  <div><button @click="change">修改年龄</button></div>
</template>

<script>
import { computed, ref } from 'vue'
export default {
  setup() {
    const age = ref(24)
    
    const newAge = computed(() => age.value + 1)

    const change = () => {
      age.value++
    }

    return {
      age,
      newAge,
      change,
      
    }
  }

}
</script>

传入对象

<template>
  <h3>今年{{age}}岁</h3>
  <h3>明年{{newAge}}岁</h3>
  <input type="text" v-model="newAge">
</template>

<script>
import { computed, ref } from 'vue'
export default {
  setup() {
    const age = ref(24)
    
    const newAge = computed({
      get() {
        return age.value + 1
      },

      set(val) {
        age.value = val - 1
      }
    })


    return {
      age,
      newAge     
    }
  }

}
</script>

watch

<template>
  <h3>{{ obj.name }}</h3>
  <button @click="updateName">修改名字</button>
</template>

<script>
import { reactive, watch } from 'vue'
export default {
  setup() {
    const obj = reactive({
      name: '张三',
      age: 18
    })

    const updateName = () => {
      obj.name = '李四'
    }

    watch(obj, (newVal, oldVal) => {
      console.log(newVal, oldVal);
    }, {
      deep: true, //深度监听 监听对象里的对象需要开启
      immediate: true // 初始化完成监听一次
    })

    return {
      obj,
      updateName
    }
  }
}
</script>

ref属性

获取元素真实dom

<template>
  <div ref="box">我是box</div>

<script>
import { onMounted, ref } from 'vue'
export default {
  setup() {
    const box = ref(null)

    onMounted(() => {
      console.log(box.value);
    }) 

    return  {
      box
    }
  }
}
</script>

子传父

son.vue

<template>
  <h2>子组件</h2>
  <div>{{money}}</div>
  <button @click="changMoney">花费50</button>
</template>

<script>
export default {
  props: ['money'],
  emits: ['changMoney'], // 声明发出的事件
  // 解构emit
  setup(props, { emit }) {
    const changMoney = () => {
      emit('changMoney', 50)
    }

    return {
      changMoney
    }
  }
}
</script>

App.vue

<template>
  <h2>父组件</h2>
  <hr>
  <son :money="money" @changMoney="changMoney"></son>
</template>

<script>
import son from '@/components/son.vue'
import { ref } from 'vue'
export default {
  components: {
    son
  },
  setup() {
    const money = ref(100)

    const changMoney = (newMoney) => {
      money.value -= newMoney
    }

    return {
      money,
      changMoney
    }
  }
}
</script>

依赖注入

  • provide 向后代组件发送数据或者函数
  • inject 接受祖先组件传来的数据或者函数

App.vue

<template>
  <h2>父组件</h2>
  <hr>
  <son></son>
</template>

<script>
import son from '@/components/son.vue'
import { provide, ref } from 'vue'
export default {
  components: {
    son
  },
  setup() {
    const money = ref(100)
    // 使用provide向后代共享数据
    provide('money', money)
    // 使用provide向后代共享函数
    const changeMoney = (val) => {
      money.value -= val
    }
    provide('changeMoney', changeMoney)
    return {
      money
    }
  }
}
</script>

son.vue

<template>
  <h2>子组件</h2>
  <div>{{money}}</div>
  <button @click="updateMoney">花费50</button>
</template>

<script>
import { inject } from 'vue'
export default {
  setup() {
    // 使用inject进行接收
    const money = inject('money')
    // 接收来自祖先组件的函数
    const changeMoney = inject('changeMoney')
    // 调用修改money
    const updateMoney = () => {
      changeMoney(50)
    }
    return {
      money,
      updateMoney
    }
  }
}
</script>

export default {

mixin

代码逻辑混入

基本使用

mixin/demoMixin.js

export const demoMixin = {
  data() {
    return {
      title: '我是混入'
    }
  },

  created() {
    console.log(this.title)
  },
}

App.vue

<script>
import { demoMixin } from './mixin/demoMixin'
export default {
  mixins: [demoMixin]
}
</script>

组件-混入合并冲突

  • 数据以组件为准
  • 生命周期都会调用
  • 函数以组件为准

全局混入

mixin/demoMixin.js

export const demoMixin = {
  data() {
    return {
      title: '我是全局混入'
    }
  },

  created() {
    console.log(this.title)
  },
}

main.js

import { createApp } from 'vue'
import App from './App.vue'
import { demoMixin } from '@/mixin/demoMixin'
const app = createApp(App)

app.mixin(demoMixin)
