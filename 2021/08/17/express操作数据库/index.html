express操作数据库

express-MySQLexpress

创建连接

// 引入mysql 第三方模块
const mysql = require('mysql');

// 创建数据库连接关系
const db = mysql.createPool({
  // 配置连接信息
  host: '127.0.0.1',
  user: 'root',
  password: '123456',
  database: 'bignews'
});

查询数据

// 查询数据
db.query('select * from users', (err, res) => {
  if(!err) {
    // 打印数据
    console.log(res);
  }
})

插入数据

方式一（推荐）

// 定义sql语句 ?表示占位符
const sql = 'insert into users set ?';
// 插入数据的属性名需要和数据库的字段名一致
const obj = {
  username: '张三',
  nickname: '小张'
}
// 执行语句
db.query(sql, obj, (err, res) => {
  if(err) {
    console.log(err);
    return;
  }

  if(res.affectedRows === 1) console.log('数据插入成功');
})

方式二

// 定义sql语句 ?表示占位符
const sql = 'insert into users (username, nickname) values (?, ?)';
// 定义要插入的数据
const obj = {
  username: '张三',
  nickname: '小张'
}
// 执行语句
db.query(sql,[obj.username, obj.nickname], (err, res) => {
  if(err) {
    console.log(err);
    return;
  }

  if(res.affectedRows === 1) console.log('数据插入成功');
})

更新数据

方式一（推荐）

const sql = 'update users set ? where id = ?';
db.query(sql, [obj, obj.id], (err, res) => {
  if(err) {
    console.log(err);
    return;
  }

  if(res.affectedRows === 1) console.log('数据修改成功');
})

方式二

const sql = 'update users set username = ?, nickname = ? where id = ?';
db.query(sql, [obj.username, obj.nickname, obj.id], (err, res) => {
  if(err) {
    console.log(err);
    return;
  }

  if(res.affectedRows === 1) console.log('数据修改成功');
})

删除数据

const sql = 'delete from users where id = ?';
db.query(sql, 4, (err, res) => {
  if(err) {
    console.log(err);
    return;
  }

  if(res.affectedRows === 1) console.log('数据删除成功');
})
